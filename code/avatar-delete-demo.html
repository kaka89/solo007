技术实现示例

// React 组件示例：删除头像按钮 function AvatarDeleteButton({ userId }) { const [isDeleting, setIsDeleting] = useState(false); const [showConfirm, setShowConfirm] = useState(false); const handleDelete = async () => { setIsDeleting(true); try { await deleteAvatar(userId); showToast('头像已成功删除', 'success'); } catch (error) { showToast(error.message, 'error'); } finally { setIsDeleting(false); setShowConfirm(false); } }; return ( <> <button onClick={() => setShowConfirm(true)} disabled={isDeleting}> {isDeleting ? '删除中...' : '删除头像'} </button> <ConfirmationDialog isOpen={showConfirm} onClose={() => setShowConfirm(false)} onConfirm={handleDelete} title="确认删除头像" message="删除后您的头像将恢复为默认头像，此操作不可撤销。" /> </> ); }
